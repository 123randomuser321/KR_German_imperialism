### Event for German subjects recieving infantry equipment

country_event = {
	id = gerimperialism.1
	title = gerimperialism.1.t
	desc = gerimperialism.1.d

	is_triggered_only = yes

	option = { # GER sends 500 base inf equipment to the subject
	name = gerimperialism.1.a
	GER = {
	send_equipment = {
	equipment = infantry_equipment_0
	amount = 500 # maybe change later
	target = ROOT
	}
	}
	}
}
